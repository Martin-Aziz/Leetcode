```python
def findKthLargest(nums, k):
    heap = nums[:k]
    heapq.heapify(heap)
```

```python
import random

def findKthLargest(nums, k):
    k = len(nums) - k
    
    def select(l, r):
        pivot = random.randint(l, r)
        nums[pivot], nums[r] = nums[r], nums[pivot]
        
        i = l
        for j in range(l, r):
            if nums[j] < nums[r]:
                nums[i], nums[j] = nums[j], nums[i]
                i += 1
        
        nums[i], nums[r] = nums[r], nums[i]
        
        if i < k:
            return select(i + 1, r)
        elif i > k:
            return select(l, i - 1)
        return nums[k]
    
    return select(0, len(nums) - 1)
```
